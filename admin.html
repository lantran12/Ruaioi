Đăng truyện mới

window.submitNewStory = function() { const id = document.getElementById('adminId').value.trim(); const title = document.getElementById('adminTitle').value.trim(); const author = document.getElementById('adminAuthor').value.trim(); const img = document.getElementById('adminImg').value.trim(); const tagsRaw = document.getElementById('adminTags').value.trim(); // Chuyển tags từ chuỗi "A, B" thành mảng ["A", "B"] const tags = tagsRaw.split(',').map(t => t.trim()); if (!id || !title) { alert("Thiếu ID hoặc Tên truyện rồi chị ơi!"); return; } const newStoryRef = ref(db, 'stories/' + id); set(newStoryRef, { title: title, author: author, img: img, tags: tags, status: "updating", updatedAt: Date.now() }).then(() => { alert("Đăng truyện mới thành công! Web sẽ tự cập nhật luôn ạ! 🚀"); // Xóa sạch form sau khi đăng xong location.reload(); }).catch(err => alert("Lỗi rồi: " + err.message)); };
