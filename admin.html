import { db } from "./firebase.js"; // Import từ file cấu hình của chị import { ref, set } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-database.js"; window.publishStory = function() { // 1. Lấy dữ liệu từ form const title = document.getElementById('adminTitle').value; const author = document.getElementById('adminAuthor').value; const img = document.getElementById('adminImg').value; const tags = document.getElementById('adminTags').value; const status = document.getElementById('adminStatus').value; // Kiểm tra cơ bản if (!title || !author) { alert("Chị ơi, điền tên tác phẩm và tác giả vào đã nhé!"); return; } // 2. Tạo ID duy nhất cho truyện (không dấu, cách bằng dấu gạch ngang) const storyId = title.toLowerCase() .normalize("NFD").replace(/[\u0300-\u036f]/g, "") // Bỏ dấu tiếng Việt .replace(/[^a-z0-9]+/g, '-'); // 3. Đẩy dữ liệu lên Firebase const storyRef = ref(db, 'stories/' + storyId); set(storyRef, { title: title, author: author, img: img, tags: tags.split(',').map(tag => tag.trim()), // Tách chuỗi thành mảng status: status, createdAt: Date.now() }) .then(() => { alert("Phát hành thành công bộ truyện: " + title + " ✨"); // Xóa sạch form sau khi đăng document.querySelector('.admin-form').reset(); }) .catch((error) => { console.error("Lỗi đăng truyện: ", error); alert("Có lỗi xảy ra, chị kiểm tra lại nhé!"); }); };
